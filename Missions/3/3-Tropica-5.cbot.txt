extern void object::Tropica5()
{//various
	object item;
	int i = 0;
	float range = 0;
	int errcode = 0;
	
	
	if(category ==WheeledGrabber)
	{
		for(int i = 0; i < 2; i++)
		{
//2) Build a converter near the chunks of titanium ore. You'll find them in the vicinity of the spaceship to the east.
//3) Produce two cubes.
			item = radar(TitaniumOre);
			goto(item.position);
			grab();
			
			item = radar(Converter);
			if(item == null) message("Waiting for Conveter to be bulid.");
			item = radar(Converter);
			if(item == null) message("Waiting to building Converter");
			while(item == null) item = radar(Converter);
			errcode = goto(item.position);
			while(errcode != 0) errcode = goto(item.position);
			drop();
			move(-2.5);
			
			while(item == null) item = radar(Titanium);
			errcode = goto(item.position);
			while(errcode != 0) errcode = goto(item.position);
			wait(15);
			grab();
			
			goto(space(position)); 
			drop();               
		}
		item = radar(Titanium);
		goto(item.position);
		grab();
		
		item = radar(BotFactory);
		if(item == null) message("Waiting for Botfactory to be built.");
		while(item == null) item = radar(BotFactory);
		errcode = goto(item.position);
		while(errcode != 0)
		{
			turn(direction(item.position));
			move(distance2d(position, item.position));
			errcode = goto(item.position);
		}
		drop();
		message("Titanium loaded into BotFactory");
		
		item = radar(PowerCell);
		goto(item.position);
		grab();

		if(item == null) message("Waiting for Subber to be bulid.");
		while(item == null) item = radar(Subber);
		errcode = goto(item.position);
		drop();
		
		item = radar(SpaceShip);
		goto(item.position);
	}
	else //Instructions for Subber
	{
		move(-5);
		
		item = radar(BlackBox);
		goto(item.position);
		grab();
		
		item = radar(SpaceShip);
		goto(item.position);
		drop();
	}
	message(category + ": Mission Complete.");
}
